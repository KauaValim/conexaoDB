DROP DATABASE IF EXISTS bdcrud;

CREATE DATABASE bdcrud;
USE bdcrud;
CREATE TABLE usuarios(id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(80) NOT NULL, 
    sexo VARCHAR(1) NOT NULL, 
    fone VARCHAR(15) NOT NULL, 
    email VARCHAR(255) NOT NULL UNIQUE, 
    senha VARCHAR(255) NOT NULL,
    adm INT NOT NULL,
    foto VARCHAR(255));

ALTER TABLE usuarios ADD COLUMN codigo_verificacao VARCHAR(10) DEFAULT NULL;

CREATE TABLE noticias (
    idnot INT AUTO_INCREMENT PRIMARY KEY,
    idusu INT,
    data DATE,
    titulo VARCHAR(255) NOT NULL,
    noticia TEXT NOT NULL,
    foto VARCHAR(255),
    FOREIGN KEY (idusu) REFERENCES usuarios(id)
);
